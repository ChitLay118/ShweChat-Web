    function sendActiveMessage() {
      const input = document.getElementById("chat-message-input");
      const text = input.value.trim();
      if (!text || !activeChatTarget) return;

      const me = auth.currentUser;
      const target = activeChatTarget;
      const conversationId = getConversationId(me.uid, target.uid);
      const timestamp = firebase.firestore.FieldValue.serverTimestamp();

      const messageObj = {
        senderId: me.uid,
        receiverId: target.uid,
        content: text,
        timestamp: timestamp
      };

      input.value = "";

      const messagesRef = db.collection('artifacts').doc(appId).collection('public').doc('data').collection('Conversations').doc(conversationId).collection('Messages');
      messagesRef.add(messageObj)
        .then(() => {
          const batch = db.batch();

          const myRecentRef = db.collection('artifacts').doc(appId).collection('users').doc(me.uid).collection('RecentChats').doc(target.uid);
          batch.set(myRecentRef, {
            uid: target.uid,
            name: target.name,
            lastMessage: text,
            timestamp: timestamp,
            fcmToken: target.fcmToken || ""
          }, { merge: true });

          const targetRecentRef = db.collection('artifacts').doc(appId).collection('users').doc(target.uid).collection('RecentChats').doc(me.uid);
          batch.set(targetRecentRef, {
            uid: me.uid,
            name: currentUserData.name,
            lastMessage: text,
            timestamp: timestamp,
            fcmToken: currentUserData.fcmToken || ""
          }, { merge: true });

          return batch.commit();
        })
        .then(() => {
          if (target.fcmToken) {
            triggerNativeNotification(target.fcmToken, currentUserData.name, text);
          }
        })
        .catch(err => {
          console.error("Transmission error:", err);
          showToast("မက်ဆေ့ခ်ျ ပို့၍မရပါ");
        });
    }

    document.getElementById("chat-message-input").addEventListener("keypress", (e) => {
      if (e.key === "Enter") {
        sendActiveMessage();
      }
    });

    // Helper utilities
    function escapeHTML(str) {
      return str.replace(/[&<>'"]/g, 
        tag => ({
          '&': '&amp;',
          '<': '&lt;',
          '>': '&gt;',
          "'": '&#39;',
          '"': '&quot;'
        }[tag] || tag)
      );
    }

    function formatTime(date) {
      let hours = date.getHours();
      let minutes = date.getMinutes();
      const ampm = hours >= 12 ? 'PM' : 'AM';
      hours = hours % 12;
      hours = hours ? hours : 12;
      minutes = minutes < 10 ? '0'+minutes : minutes;
      return `${hours}:${minutes} ${ampm}`;
    }
